⚙️ Sheet Sync Setup

This app syncs new registrations from your Google Sheet. Since the sheet contains PII, a Google Apps Script proxy runs as you and returns only group names + emojis — no PII ever leaves the sheet.

Column X of the sheet stores each group's emoji. Run fillEmojis() once to auto-suggest them, then edit freely in the sheet. The web app picks up whatever you put there.

Step 1. Open your spreadsheet → Extensions → Apps Script. Replace all code with this, then save:

// ── Column indices (0-based) ────────────────────────── const NAME_COL = 4; // E: Group name const DESC_COL = 8; // I: Brief description // Emoji column: found dynamically by header (any cell containing "emoji") function getEmojiCol(sheet) { const headers = sheet.getRange(1, 1, 1, sheet.getLastColumn()).getValues()[0]; const idx = headers.findIndex(h => String(h).toLowerCase().includes('emoji')); return idx >= 0 ? idx : 23; // falls back to col X if header not found } // ── Keyword → emoji suggestions ─────────────────────── const KEYWORD_MAP = [ [/electric|electrical|kme/i, '⚡'], [/fire.?(truck|engine|dept|department|station)|zayante/i, '🚒'], [/tractor|tractors|agriculture|agricultural|ag.?history/i,'🚜'], [/motorcycle|motorcycles|biker|harley|pharoah|riders.?mc/i,'🏍️'], [/marine|army|navy|military|veteran|color.?guard|korean|vietnam|remembrance|memorial|corporal/i,'🎖️'], [/recycle|recycling|greenwaste|green.?waste|garbage|trash/i,'♻️'], [/museum|historical.?society|history.?project|heritage/i, '🏛️'], [/veterinary|mobile.?vet|beneath.?the.?redwood/i, '🐾'], [/lumber|hardware|home.?center/i, '🪚'], [/jazz|saxophone/i, '🎷'], [/band|marching.?band|orchestra/i, '🥁'], [/scout/i, '🏕️'], [/little.?league|baseball/i, '⚾'], [/softball/i, '⚾'], [/football|cheer/i, '🏈'], [/horse|horses|equestrian/i, '🐎'], [/train|railroad|railway|locomotive|roaring.?camp/i, '🚂'], [/art.?car|virgie/i, '🎨'], [/van|vans|vivid|vanagon/i, '🌈'], [/rotary/i, '🔔'], [/corvette|mustang|gto|camaro|studebaker|biscayne|citroen|lotus|porsche|car.?club|classic.?car/i,'🚗'], [/dog|canine/i, '🐕'], ]; function suggestEmoji(name, desc) { const text = (name + ' ' + (desc || '')); for (const [re, em] of KEYWORD_MAP) { if (re.test(text)) return em; } return '🎪'; } // ── Run this once to fill column X for all existing rows ── // Re-run any time to fill newly empty cells (won't overwrite edits). function fillEmojis() { const sheet = SpreadsheetApp.getActiveSpreadsheet() .getSheetByName('form responses copy'); if (!sheet) { Logger.log('Sheet not found'); return; } const emojiCol = getEmojiCol(sheet); const data = sheet.getDataRange().getValues(); let filled = 0; for (let i = 1; i < data.length; i++) { const existing = String(data[i][emojiCol] || '').trim(); if (existing) continue; // don't overwrite manual edits const name = String(data[i][NAME_COL] || '').trim(); const desc = String(data[i][DESC_COL] || '').trim(); if (!name || name.toLowerCase() === 'n/a') continue; sheet.getRange(i + 1, emojiCol + 1).setValue(suggestEmoji(name, desc)); filled++; } Logger.log('Filled ' + filled + ' emoji cells.'); } // ── Web app endpoint (deploy as: Execute as Me, Anyone) ── function doGet(e) { const sheet = SpreadsheetApp.getActiveSpreadsheet() .getSheetByName('form responses copy'); if (!sheet) return out({ error: 'Sheet not found' }, e); const emojiCol = getEmojiCol(sheet); const rows = sheet.getDataRange().getValues(); const result = []; for (let i = 1; i < rows.length; i++) { const ts = rows[i][0]; const name = String(rows[i][NAME_COL] || '').trim(); const emoji = String(rows[i][emojiCol] || '').trim(); if (!name || name.toLowerCase() === 'n/a') continue; const tsStr = ts instanceof Date ? Utilities.formatDate(ts, Session.getScriptTimeZone(), 'M/d/yyyy H:mm:ss') : String(ts); result.push({ name, ts: tsStr, emoji: emoji || null }); } return out({ rows: result }, e); } function out(data, e) { const json = JSON.stringify(data); const cb = e && e.parameter && e.parameter.callback; return ContentService .createTextOutput(cb ? cb + '(' + json + ');' : json) .setMimeType(cb ? ContentService.MimeType.JAVASCRIPT : ContentService.MimeType.JSON); }

Step 2. Make sure cell X1 has a header containing the word emoji (any capitalization — e.g. "Emoji reference", "Parade Emoji", etc.).

Then back in Apps Script, click Run → fillEmojis. Authorize when prompted. This fills column X with suggested emojis for all existing rows. You can edit any cell in column X directly in the sheet — those edits are what the web app uses.

Step 3. Click Deploy → New deployment (or Manage deployments → edit → update to new version if you already deployed). Set type to Web app, execute as Me, access Anyone. Deploy and authorize.
